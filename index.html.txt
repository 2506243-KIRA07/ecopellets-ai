<script>
function getRecommendation() {
    let soil = document.getElementById("soil").value;
    let crop = document.getElementById("crop").value;
    let pest = document.getElementById("pest").value;

    if (!soil || !crop || !pest) {
        document.getElementById("result").innerHTML =
            "⚠️ Please select soil, crop, and pest type.";
        return;
    }

    document.getElementById("result").innerHTML =
        `<b>AI Recommendation:</b><br>
        Use EcoPellets bio-fertilizer suitable for <b>${soil}</b> soil.<br>
        Recommended for <b>${crop}</b> crops.<br>
        Effective organic pest control for <b>${pest}</b>.`;
}
</script>
